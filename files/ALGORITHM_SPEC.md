        else:

        twist = calculate_torso_twist(landmarks)

        knee_angle = calculate_knee_angle(landmarks)

def draw_spider_chart(frame, metrics, config=SPIDER_CONFIG):
    """
    –û—Ç—Ä–∏—Å–æ–≤–∫–∞ —Ä–∞–¥–∞—Ä–Ω–æ–π –¥–∏–∞–≥—Ä–∞–º–º—ã –Ω–∞ –∫–∞–¥—Ä–µ.
    
    metrics: —Å–ª–æ–≤–∞—Ä—å {metric_name: value (0-100)}
    """
    center_x, center_y = config["position"]
    center_x += config["radius"]
    center_y += config["radius"]
    radius = config["radius"]
    
    num_axes = len(SPIDER_AXES)
    angle_step = 2 * math.pi / num_axes
    
    # –°–æ–∑–¥–∞—ë–º overlay –¥–ª—è –ø—Ä–æ–∑—Ä–∞—á–Ω–æ—Å—Ç–∏
    overlay = frame.copy()
    
    # –†–∏—Å—É–µ–º —Ñ–æ–Ω (–∫—Ä—É–≥)
    cv2.circle(overlay, (center_x, center_y), radius, (40, 40, 40), -1)
    
    # –†–∏—Å—É–µ–º –æ—Å–∏ –∏ –ø–æ–¥–ø–∏—Å–∏
    points = []
    for i, axis in enumerate(SPIDER_AXES):
        angle = -math.pi/2 + i * angle_step  # –Ω–∞—á–∏–Ω–∞–µ–º —Å–≤–µ—Ä—Ö—É
        
        # –ö–æ–Ω–µ—Ü –æ—Å–∏
        end_x = center_x + radius * math.cos(angle)
        end_y = center_y + radius * math.sin(angle)
        
        # –†–∏—Å—É–µ–º –æ—Å—å
        cv2.line(overlay, (center_x, center_y), (int(end_x), int(end_y)), 
                 config["line_color"], 1)
        
        # –ü–æ–¥–ø–∏—Å—å
        label_x = center_x + (radius + 15) * math.cos(angle)
        label_y = center_y + (radius + 15) * math.sin(angle)
        cv2.putText(overlay, axis["short"], (int(label_x) - 10, int(label_y) + 5),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.4, (255, 255, 255), 1)
        
        # –¢–æ—á–∫–∞ –∑–Ω–∞—á–µ–Ω–∏—è
        value = metrics.get(axis["name"], 50) / 100  # –Ω–æ—Ä–º–∞–ª–∏–∑—É–µ–º –∫ 0-1
        point_x = center_x + radius * value * math.cos(angle)
        point_y = center_y + radius * value * math.sin(angle)
        points.append((int(point_x), int(point_y)))
    
    # –†–∏—Å—É–µ–º –∑–∞–ø–æ–ª–Ω–µ–Ω–Ω—É—é –æ–±–ª–∞—Å—Ç—å
    if len(points) > 2:
        pts = np.array(points, np.int32)
        cv2.fillPoly(overlay, [pts], config["fill_color"])
        cv2.polylines(overlay, [pts], True, (255, 255, 255), 2)
    
    # –†–∏—Å—É–µ–º —Ç–æ—á–∫–∏ –∑–Ω–∞—á–µ–Ω–∏–π
    for point in points:
        cv2.circle(overlay, point, 4, (255, 255, 255), -1)
    
    # –û–±—â–∏–π –±–∞–ª–ª –≤–Ω–∏–∑—É
    avg_score = sum(metrics.values()) / len(metrics) if metrics else 0
    cv2.putText(overlay, f"Score: {int(avg_score)}", 
                (center_x - 25, center_y + radius + 35),
                cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 255), 1)
    
    # –ü—Ä–∏–º–µ–Ω—è–µ–º –ø—Ä–æ–∑—Ä–∞—á–Ω–æ—Å—Ç—å
    cv2.addWeighted(overlay, config["background_alpha"], frame, 
                    1 - config["background_alpha"], 0, frame)
    
    return frame
